# auto_ship（自动发货）Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 Temu 商家中心发货单列表「待装箱发货」tab 一键遍历所有发货单，对非本地仓单自动走完「打印打包标签(先发货后打印)→等包裹号→批量装箱发货→编辑页填箱子和袋子+1箱→确认发货(人工门控)」，本地仓单跳过，单步失败记录并跳过继续下一单。

**Architecture:** 纯 content-script 单页编排（无 background、无 native host）。三层职责：① 纯逻辑模块 `auto_ship-logic.js`（仓库判定/包裹号判定/去重/汇总，node 单测）；② DOM 适配层（index.js 内，把所有易变 selector + 读/写/点动作隔离成稳定接口，基于 dump 的真实 DOM 落实）；③ 编排层（index.js 内主循环 + 单发货单状态机，按 DOM 适配层接口写，控制流精确）。循环运行态驻留 content script 内存（SPA 不整页 reload），仅「自动确认发货」开关存 `chrome.storage.local`。

**Tech Stack:** Chrome MV3 content script（vanilla JS, IIFE + `window.AgentSeller` core API）、`chrome.storage.local`、Node 内置 test runner（`node --test`）做纯函数单测。Temu = Beast UI（`data-testid` + 文字匹配优先，禁 hash class）。

设计依据：`docs/superpowers/specs/2026-05-28-auto-ship-design.md`

## Open Issues 的决议（spec §13，dump 后在对应 task 内最终确认）

1. **发货单号 ↔ 分组 checkbox**：先假设「一个发货单号 = 一个 rowspan 分组 = 一个分组级 checkbox」（沿用 packing_label 表格结构）。Task 1 dump ① 后确认；若一个发货单跨多个物流分组，Task 4 的 `findGroup`/`selectGroup` 需调整为「选中该发货单号下所有分组 checkbox」。
2. **in-page tab 刷新方式**：先假设「点 tab 文字切换会触发后台重新拉数」。Task 1 dump ① 捕获 tab bar DOM；Task 4 `refreshViaTabSwitch` 实现为「切到另一 tab → 等 → 切回 → 等表格」；若另有显式刷新按钮，dump 后改用。
3. **包裹号轮询超时**：点完「先发货后打印」小弹窗后等包裹号生成，超时 `15000ms`（中途 `7500ms` 时切 tab 刷新一次重读）；超时计「业务：包裹号超时未生成」失败、跳过。
4. **空包裹号判定**：`isValidPackageNo` 兼容空串 / `-` / `—` / `无`/`待生成`/`暂无`/`未生成`；Task 1 dump ① 后用真实空值表现校准 `auto_ship-logic.js` 的占位符正则。

```js
// 纯函数：本地仓判定 / 包裹号有效判定 / 发货单号去重 / 结果汇总。
// 格式无关——DOM 单元格的原始文本提取放 index.js DOM 适配层。
// 双用途：浏览器挂 window.__AutoShipLogic；node 测试用 module.exports。
(function () {
  'use strict';

  const LOCAL_WAREHOUSE = '化州中正科技';
  // 包裹号占位符（无有效值）。Task 1 dump 后按真实空值表现增补。
  const PKG_PLACEHOLDERS = ['', '-', '—', '无', '待生成', '暂无', '未生成'];

  // 仓库名（已由 DOM 层提取出的纯名字）是否本地仓。
  function isLocalWarehouse(name) {
    return String(name == null ? '' : name).includes(LOCAL_WAREHOUSE);
  }

  // 包裹号值（已由 DOM 层提取）是否为有效包裹号。
  function isValidPackageNo(value) {
    const v = String(value == null ? '' : value).trim();
    if (!v) return false;
    if (PKG_PLACEHOLDERS.includes(v)) return false;
    if (/^[-—]+$/.test(v)) return false;
    return true;
  }

  // 去重 + 去空 + trim，保持首次出现顺序。
  function dedupOrderNos(list) {
    const seen = new Set();
    const out = [];
    for (const x of list || []) {
      const k = String(x == null ? '' : x).trim();
      if (k && !seen.has(k)) { seen.add(k); out.push(k); }
    }
    return out;
  }

  // 汇总文案：「处理 X 单 / 跳过本地仓 Y / 失败 Z」+ 失败明细行。
  function summarize({ shipped, skippedLocal, fails }) {
    const f = fails || [];
    let s = `处理 ${shipped} 单 / 跳过本地仓 ${skippedLocal} / 失败 ${f.length}`;
    if (f.length) {
      s += '\n' + f.map((x) => `${x.orderNo}｜${x.step}｜${x.reason}`).join('\n');
    }
    return s;
  }

  const api = { LOCAL_WAREHOUSE, PKG_PLACEHOLDERS, isLocalWarehouse, isValidPackageNo, dedupOrderNos, summarize };
  if (typeof window !== 'undefined') window.__AutoShipLogic = api;
  if (typeof module !== 'undefined' && module.exports) module.exports = api;
})();
```

  // ── 易变选择器集中区（对照 samples/ 落实/调整）──
  const SEL = {
    dataRow: '[data-testid="beast-core-table-body-tr"]',   // 兜底 'tbody tr'
    groupCheckbox: '[data-testid="beast-core-checkbox"]',   // 分组级 checkbox（rowspan 首行）
    tab: '[role="tab"]',                                    // in-page tab
    scrollContainer: '[class*="scroll-view"],[class*="table-body"]', // 虚拟滚动容器
  };
  const TAB_PENDING = '待装箱发货';
  const TAB_RECEIVED = '待仓库收货';

  function dataRows() {
    let rows = Array.from(document.querySelectorAll(SEL.dataRow));
    if (!rows.length) rows = Array.from(document.querySelectorAll('tbody tr'));
    return rows;
  }
  function groupCheckbox(tr) {
    return tr ? tr.querySelector(SEL.groupCheckbox) : null;
  }

  // 从分组首行提取发货单号。正则对照 samples/table_and_tabs.txt 落实（可能是独立列而非 label）。
  function readOrderNo(firstTr) {
    if (!firstTr) return '';
    const m = U.normText(firstTr.textContent).match(/发货单号[:：]?\s*([A-Za-z0-9]+)/);
    return m ? m[1] : '';
  }
  // 发货仓库（分组级，在「发货信息」列/首行）。正则对照 samples 落实。

    const t = U.normText(group.firstTr.textContent);
    const m = t.match(/发货仓库[:：]?\s*([^\s，,；;:：]+)/);
    return m ? m[1] : '';
  }
  // 包裹号（「包裹号」列）。空值表现对照 samples 落实（交给 logic.isValidPackageNo 判定）。
  function readPackageNo(group) {
    const t = U.normText(group.firstTr.textContent);
    const m = t.match(/包裹号[:：]?\s*([^\s，,；;:：]*)/);
    return m ? m[1] : '';
  }

  // ── in-page tab ──
  function tabEls() { return Array.from(document.querySelectorAll(SEL.tab)); }

    const el = tabEls().find((t) => t.getAttribute('aria-selected') === 'true'
      || /active|selected/.test(t.className));
    return el ? U.normText(el.textContent) : '';
  }

  function catLabel(cat) { return cat === 'data' ? '校验' : cat === 'biz' ? '业务' : '读取'; }

  // 取下一个未处理发货单号；本轮无 → 切 tab 刷新再确认一次（防刷新延迟）。
  async function nextPick(processed) {
    await ensureOnPendingTab();
    let live = await scanOrderNos();
    let pick = live.find((no) => !processed.has(no));
    if (pick) return pick;
    await refreshViaTabSwitch();
    live = await scanOrderNos();
    return live.find((no) => !processed.has(no)) || null;
  }

  async function runLoop() {
    run.active = true; run.stopRequested = false;
    setSummary('');
    const processed = new Set();
    let shipped = 0, skippedLocal = 0;
    const fails = [];
    try {
      await ensureOnPendingTab();
      const total = (await scanOrderNos()).length;     // 进度参考（实际以每轮重扫为准）
      while (!run.stopRequested) {
        const orderNo = await nextPick(processed);
        if (!orderNo) break;
        const denom = Math.max(total, processed.size + 1);
        setProgress(`处理中 ${processed.size + 1}/${denom}\n当前发货单：${orderNo}`);
        try {
          const r = await processOrder(orderNo);
          if (r.kind === 'local') skippedLocal += 1;
          else if (r.kind === 'shipped') shipped += 1;
          // cancelled：不计 shipped，仍算已处理
        } catch (err) {
          fails.push({ orderNo, step: catLabel(err && err._cat), reason: (err && err.message) || String(err) });
          console.warn('[auto_ship] 单失败:', orderNo, err);
          try { await closeEditPage(); } catch (_) {}   // 清残留弹窗，避免污染下一单
        }
        processed.add(orderNo);
      }
    } finally {
      run.active = false;
      setRunningUI(false);
      setProgress(run.stopRequested ? '已停止' : '已完成');
      setSummary(L.summarize({ shipped, skippedLocal, fails }));
      AS.showToast(`自动发货结束：发 ${shipped} / 跳过本地仓 ${skippedLocal} / 失败 ${fails.length}`,
